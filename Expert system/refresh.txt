import tkinter as tk

def refresh_checkbox():
    var.set(not var.get())  # Update the variable (toggle its value)
    checkbox.update()  # Refresh the checkbox appearance

# Create the main window
window = tk.Tk()

# Create a variable to store the checkbox state
var = tk.BooleanVar()

# Create the checkbox
checkbox = tk.Checkbutton(window, text="Check me", variable=var)
checkbox.pack()

# Create a button to refresh the checkbox
refresh_button = tk.Button(window, text="Refresh", command=refresh_checkbox)
refresh_button.pack()

# Start the Tkinter event loop
window.mainloop()
